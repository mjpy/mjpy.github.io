Takes document number as argument

Firefox

from selenium import webdriver
import sys

url = "https://checkwpstatus.mom.gov.sg/Pages/home.aspx"
doc_nr = sys.argv[1]

driver = webdriver.Firefox()
driver.get(url)

driver.find_element_by_id("ctl00_ctl40_g_6970c440_5b2c_4b60_ad5e_3fc670e7fb1f_txtCsn").send_keys(doc_nr)
driver.find_element_by_id("ctl00_ctl40_g_6970c440_5b2c_4b60_ad5e_3fc670e7fb1f_btnSubmit").click()

output = driver.find_element_by_id('ctl00_ctl40_g_6970c440_5b2c_4b60_ad5e_3fc670e7fb1f_cellDateValue')
print(output.get_attribute('innerHTML'))
driver.close()
